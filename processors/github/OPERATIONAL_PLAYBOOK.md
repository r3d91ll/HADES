## 3. Update & Lifecycle Management

### Cascading Deactivation

```python
def deactivate_file_and_descendants(file_key: str, run_id: str):
    """Cascade deactivation to chunks and edges."""
    timestamp = datetime.now(timezone.utc).isoformat()
    
    # AQL for atomic cascading deactivation
    query = """
    LET file_id = CONCAT("base_github_files/", @file_key)
    LET timestamp = @timestamp
    LET run_id = @run_id
    
    // Deactivate file
    UPDATE { _key: @file_key, active: false, deactivated_at: timestamp, deactivation_run: run_id }
    IN base_github_files
    
    // Find and deactivate chunks
    LET chunks = (
        FOR chunk IN base_github_chunks
        FILTER chunk.parent_key == @file_key
        RETURN chunk._id
    )
    
    FOR chunk_id IN chunks
        UPDATE { _id: chunk_id, active: false, deactivated_at: timestamp }
        IN base_github_chunks
    
    // Deactivate outbound edges from chunks
    FOR chunk_id IN chunks
        FOR e IN edges_implements
        FILTER e._from == chunk_id
        UPDATE e WITH { active: false, deactivated_at: timestamp }
        IN edges_implements
    
    RETURN { 
        file_deactivated: 1,
        chunks_deactivated: LENGTH(chunks),
        edges_deactivated: LENGTH(chunks)
    }
    """
    
    return db.aql.execute(query, bind_vars={
        'file_key': file_key,
        'timestamp': timestamp,
        'run_id': run_id
    })
```

class ActiveLearningQueue:
    def __init__(self, db, threshold_low=0.65, threshold_high=0.85):
        self.db = db
        self.threshold_low = threshold_low
        self.threshold_high = threshold_high

def handle_lfs_file(file_path: str, repo: git.Repo) -> dict:
    """Handle Git LFS pointers."""
    content = open(file_path, 'rb').read()
    
    if content.startswith(b'version https://git-lfs.github.com'):
        # Parse LFS pointer
        lines = content.decode('utf-8').split('\n')
        pointer = {}
        for line in lines:
            if ' ' in line:
                key, value = line.split(' ', 1)
                pointer[key] = value
        
        return {
            'is_lfs': True,
            'lfs_oid': pointer.get('oid'),
            'lfs_size': int(pointer.get('size', 0)),
            'skip_content': True
        }
    
    # Regular file size check
    size = len(content)
    return {
        'is_lfs': False,
        'size_bytes': size,
        'skip_content': size > 10_000_000  # 10MB limit
    }

## 6. Embeddings & Infrastructure

### Model Management

```python
class EmbeddingManager:
    CURRENT_VERSION = "1.0.0"
    MAX_REEMBED_RATE = 1000  # chunks per minute
    
    def needs_reembedding(self, chunk: dict) -> bool:
        """Check if chunk needs new embedding."""
        return (
            chunk.get('embedding_version') != self.CURRENT_VERSION or
            chunk.get('embedding_model') != 'jinaai/jina-embeddings-v4' or
            len(chunk.get('embedding', [])) != 2048
        )
    
    def reembed_background_job(self):
        """Background re-embedding with rate limiting."""
        while True:
            # Find chunks needing update
            chunks = self.db.aql.execute("""
                FOR c IN base_github_chunks
                FILTER c.embedding_version != @version
                LIMIT @rate
                RETURN c
            """, bind_vars={'version': self.CURRENT_VERSION, 'rate': self.MAX_REEMBED_RATE})
            
            if not chunks:
                break
                
            # Batch embed with backpressure
            for batch in chunked(chunks, EMBED_BATCH_SIZE):
                if embed_queue_size() > MAX_QUEUE_SIZE:
                    time.sleep(10)  # Backpressure
                
                embeddings = self.embedder.embed_batch(batch)
                self.update_embeddings(batch, embeddings)
            
            time.sleep(60)  # Rate limit
```

### Batching & Backpressure

```python
# Configuration
EMBED_BATCH_SIZE = 128
PARSE_WORKERS = cpu_count()
EMBED_WORKERS = 2  # GPU streams
MAX_CHUNKS_PER_FILE = 10000
MAX_EMBED_QUEUE = 10000

class ProcessingPipeline:
    def __init__(self):
        self.parse_queue = Queue(maxsize=1000)
        self.embed_queue = Queue(maxsize=MAX_EMBED_QUEUE)
        self.parse_workers = []
        self.embed_workers = []
    
    def should_pause_parsing(self) -> bool:
        """Backpressure: pause if embed queue is full."""
        return self.embed_queue.qsize() > MAX_EMBED_QUEUE * 0.8
    
    def parse_file(self, file_path: str):
        """Parse with backpressure."""
        if self.should_pause_parsing():
            time.sleep(5)
            
        chunks = self.tree_sitter_chunker.chunk_file(file_path)
        
        if len(chunks) > MAX_CHUNKS_PER_FILE:
            logger.warning(f"File {file_path} has {len(chunks)} chunks, truncating")
            chunks = chunks[:MAX_CHUNKS_PER_FILE]
        
        for batch in chunked(chunks, EMBED_BATCH_SIZE):
            self.embed_queue.put(batch)
```

## 7. Security & Compliance

### Secret Scrubbing

```python
import re

SECRET_PATTERNS = [
    (r'[a-zA-Z0-9]{40}', 'github_token'),  # GitHub tokens
    (r'sk-[a-zA-Z0-9]{48}', 'openai_key'),  # OpenAI keys
    (r'AKIA[0-9A-Z]{16}', 'aws_key'),      # AWS keys
    (r'-----BEGIN (RSA )?PRIVATE KEY-----', 'private_key'),
]

def scrub_secrets(content: str) -> tuple[str, list]:
    """Remove secrets from content."""
    found_secrets = []
    scrubbed = content
    
    for pattern, secret_type in SECRET_PATTERNS:
        matches = re.finditer(pattern, content)
        for match in matches:
            found_secrets.append({
                'type': secret_type,
                'position': match.start()
            })
            scrubbed = scrubbed.replace(match.group(), f'[REDACTED_{secret_type}]')
    
    return scrubbed, found_secrets
```

### License Detection

```python
LICENSE_PATTERNS = {
    'MIT': r'MIT License|Permission is hereby granted, free of charge',
    'Apache-2.0': r'Apache License,? Version 2\.0',
    'GPL-3.0': r'GNU GENERAL PUBLIC LICENSE\s+Version 3',
    'BSD-3-Clause': r'BSD 3-Clause License|Redistribution and use in source and binary forms',
}

def detect_license(repo_path: str) -> dict:
    """Detect repository license."""
    license_files = ['LICENSE', 'LICENSE.txt', 'LICENSE.md', 'COPYING']
    
    for file_name in license_files:
        file_path = os.path.join(repo_path, file_name)
        if os.path.exists(file_path):
            content = open(file_path).read()
            for license_name, pattern in LICENSE_PATTERNS.items():
                if re.search(pattern, content, re.IGNORECASE):
                    return {
                        'license': license_name,
                        'license_file': file_name,
                        'confidence': 0.95
                    }
    
    return {'license': 'UNKNOWN', 'confidence': 0.0}
```
